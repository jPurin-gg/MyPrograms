int i,j;
GLdouble p[4][3];
p[0][0] = 1.0;
p[0][1] = p[0][2] = p[1][2] = 0.0;
p[1][0] = p[2][0] = p[3][0] = -1.0 / 3;
p[2][1] = p[3][1] = -sqrt(2.0) / 3;
p[1][1] = -2.0 * p[2][1];
p[2][2] = sqrt(6.0) / 3.0;
p[3][2] = -p[2][2];
glBegin(GL_LINES);
for (i = 0; i < 3; i++) {
    for (j = i + 1; j < 4; j++) {
        glVertex3dv(p[i]);
        glVertex3dv(p[j]);
    }
}
glEND();
